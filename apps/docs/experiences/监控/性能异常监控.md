# 性能异常监控

## 监控长任务

```js
if ('PerformanceObserver' in window) {
  const observer = new PerformanceObserver((list) => {
    list.getEntries().forEach((entry) => {
      if (entry.duration > 100) {
        // 比如超过 100ms 的长任务
        reportLongTask(entry)
      }
    })
  })

  observer.observe({ entryTypes: ['longtask'] })
}

let longTasks = []

/**
 * 这只是一个上报的思路，具体怎么上报还得看具体场景
 */
function reportLongTask(entry) {
  longTasks.push({
    startTime: entry.startTime,
    duration: entry.duration,
    name: entry.name,
  })

  // 如果最近 5 秒内有 3 个以上长任务
  const recentTasks = longTasks.filter((task) => performance.now() - task.startTime < 5000)

  if (recentTasks.length > 3) {
    report({
      averageDuration: recentTasks.reduce((sum, t) => sum + t.duration, 0) / recentTasks.length,
      type: 'frequent_long_tasks',
      tasks: recentTasks,
    })
  }

  // 只保留最近 10 秒的任务
  longTasks = longTasks.filter((task) => performance.now() - task.startTime < 10000)
}
```

## 监控内存

```js
if (performance.memory) {
  setInterval(() => {
    const used = performance.memory.usedJSHeapSize
    const total = performance.memory.totalJSHeapSize
    const usage = used / total

    // 内存使用超过 90%
    if (usage > 0.9) {
      this.report({
        type: 'high_memory_usage',
        usage: usage * 100 + '%',
        used: formatBytes(used),
        total: formatBytes(total),
      })
    }
  }, 10000)
}
```

## 监控帧率

利用 requestAnimationFrame 计算 1 秒内被执行的次数

```js
let lastTime = performance.now()
let frameCount = 0

const checkFPS = () => {
  frameCount++
  const currentTime = performance.now()
  const elapsed = currentTime - lastTime

  // 每 1 秒计算一次
  if (elapsed >= 1000) {
    const fps = Math.round((frameCount * 1000) / elapsed)

    // 帧率低于 30
    if (fps < 30) {
      this.report({
        timestamp: Date.now(),
        type: 'low_fps',
        fps,
      })
    }

    frameCount = 0
    lastTime = currentTime
  }

  requestAnimationFrame(checkFPS)
}

checkFPS()
```

## 监控卡顿

```js
let lastCheck = Date.now()

setInterval(() => {
  const now = Date.now()
  const timeDiff = now - lastCheck

  // 应该每秒检查 1 次，如果超过 2 秒，说明主线程卡住了
  if (timeDiff > 2000) {
    this.report({
      type: 'main_thread_stuck',
      stuckTime: timeDiff,
    })
  }

  lastCheck = now
}, 1000)
```
